if (FIPS_EMSCRIPTEN)
    set(slang "glsl100")
elseif (FIPS_WINDOWS)
    set(slang "hlsl4:glsl330")
elseif (FIPS_OSX)
    if (FIPS_IOS)
        set(slang "metal_ios:metal_sim")
    else()
        set(slang "metal_macos")

    set(slang "glsl330")

fips_begin_app(v6502r windowed)
    fips_vs_warning_level(3)
    fips_files(main.c v6502r.h util.c)
    fips_files(gfx.c gfx.h)
    fips_files(pick.c pick.h)
    fips_files(sim.c sim.h)
    fips_files(trace.c trace.h)
    fips_files(asm.c asm.h)
    fips_files(ui.cc ui.h)
    fips_files(ui_asm.cc ui_asm.h)
    fips_files(segdefs.c segdefs.h nodenames.c nodenames.h)
    sokol_shader(gfx.glsl ${slang})
    fips_dir(res)
    fips_generate(FROM fonts.yml TYPE dump HEADER fonts.h)
    fips_generate(FROM markdown.yml TYPE dump HEADER markdown.h)
    fips_files(help_assembler.md about.md favicon.png)
    fips_deps(sokol perfect6502 texteditor asmx)
